<Steps>
  <Step title="Get your API key">
    Go to [app.strait.dev](https://app.strait.dev) and navigate to Settings > API Keys. Create a new key with the scopes you need.
  </Step>
  <Step title="Set the environment variable">
    ```bash
    export STRAIT_API_KEY=your_api_key_here
    export STRAIT_BASE_URL=https://api.strait.dev
    ```
  </Step>
  <Step title="Verify the connection">
    ```bash
    curl -H "Authorization: Bearer $STRAIT_API_KEY" https://api.strait.dev/health
    ```
  </Step>
</Steps>
